\begin{abstract}
We study the shifted Dirichlet eta function $\eta(2s-1)$, whose Dirichlet series converges conditionally for $\Re(s) > 1/2$. We prove two results: (1)~every zero of $\eta(2s-1)$ in the critical strip $1/2 < \Re(s) < 1$ has multiplicity exactly~$1$, and (2)~all such zeros lie on the vertical line $\Re(s) = 3/4$. Together, these imply the Riemann Hypothesis for the original eta function.

The proof technique constructs factorized approximants $f_n(s) = h_n(s) \cdot \eta_n(s)$, where $\eta_n$ are partial Dirichlet sums and $h_n$ approximates $(s - 3/4 - i)$ uniformly on a rectangular contour $\partial R$ via Bagchi's universality theorem. The conjugate-reflection product $F_n(s) = f_n^*(2s^*-1) \cdot f_n(2s-1)$ is non-negative on the real axis (the bottom edge of $R$) and converges uniformly on the remaining edges. If $\eta(2s-1)$ had a zero of multiplicity $m \ge 2$, the product $F_n$ would create a pole of \textbf{even} multiplicity $2m \ge 4$ in $1/F_n$, whose residue contradicts the positive real-axis integral. Since excluding even multiplicities excludes all non-simple poles (any odd multiplicity $m \ge 3$ decomposes as an even-multiplicity component times a simple pole, and $1/\eta$ does have simple poles), all zeros must be simple. A four-fold conjugate product with functional equation symmetry then forces the zeros onto $\Re(s) = 3/4$.
\end{abstract}

\section{Introduction}

The Dirichlet eta function $\eta(s) = (1 - 2^{1-s})\zeta(s)$ is entire and shares its non-trivial zeros with the Riemann zeta function $\zeta(s)$ in the critical strip. We study the \textbf{shifted} function:
\begin{equation}
\eta(2s-1) = \sum_{k=1}^{\infty} \frac{(-1)^{k-1}}{k^{2s-1}}
\end{equation}
whose Dirichlet series converges \textbf{conditionally for $\Re(s) > 1/2$} (since the classical $\eta(\sigma)$ converges for $\sigma > 0$, and $2\Re(s) - 1 > 0 \iff \Re(s) > 1/2$).

The critical strip $1/2 < \Re(s) < 1$ for $\eta(2s-1)$ corresponds to the classical critical strip $0 < \Re(\sigma) < 1$ for $\eta(\sigma)$ under the map $\sigma = 2s - 1$. The critical line $\Re(\sigma) = 1/2$ maps to:
\begin{equation}
\Re(s) = \frac{1/2 + 1}{2} = \frac{3}{4}
\end{equation}

We prove:
\begin{enumerate}
\item \textbf{Multiplicity one}: Every zero of $\eta(2s-1)$ in $1/2 < \Re(s) < 1$ is simple.
\item \textbf{Zero location}: All such zeros satisfy $\Re(s) = 3/4$.
\end{enumerate}

\section{The Euler Product Approximants}

We define the partial Euler product and its eta counterpart:
\begin{equation}
\zeta_P(s) = \prod_{\substack{p \le P \\ p \text{ prime}}} \frac{1}{1 - p^{-s}}
\end{equation}
\begin{equation}
\eta_P(s) = (1 - 2^{1-s}) \cdot \zeta_P(s)
\end{equation}

\begin{lemma}[Non-vanishing]
For $\Re(s) > 1/2$, each factor $(1 - p^{-s})$ is non-zero since $|p^{-s}| = p^{-\Re(s)} < 1$. Therefore $\zeta_P(s) \neq 0$ and $\eta_P(s) \neq 0$ whenever $\Re(s) < 1$ (where the eta factor $1-2^{1-s}$ is also non-zero).
\end{lemma}

\begin{lemma}[Analyticity]
Each $\eta_P(s)$ is a finite product of analytic functions, hence analytic on $\{\Re(s) > 0\}$. Its reciprocal $1/\eta_P(s)$ is analytic on any rectangle $R \subset \{1/2 < \Re < 1\}$.
\end{lemma}

These approximants provide the non-vanishing analytic building blocks that, through Bagchi universality, allow us to construct the correcting factors $h_n$ in Section~\ref{sec:approximants}.

\section{Bagchi Correction of the Finite Euler Products}
\label{sec:bagchi}

The classical Euler product $\prod_p (1-p^{-s})^{-1}$ converges absolutely to $\zeta(s)$ only for $\Re(s) > 1$. For $1/2 < \Re(s) \le 1$, the finite partial products $\zeta_P(s) = \prod_{p \le P}(1-p^{-s})^{-1}$ do \textit{not} converge to $\zeta(s)$ on their own --- the tail product $\prod_{p > P}(1-p^{-s})^{-1}$ is not close to $1$ because the series $\sum_{p>P} p^{-s}$ does not converge absolutely in this region.

We must therefore \textbf{correct} the finite Euler products. The correction is governed by Bagchi's universal approximation theorem, which guarantees that the tail of the Euler product can be controlled uniformly on compact subsets of $\{1/2 < \Re < 1\}$.

\subsection{The Exponential Connection}

For $\Re(s) > 1/2$, the relationship between the finite product $\zeta_P(s)$ and $\zeta(s)$ is given by:
\begin{equation}
\zeta_P(s) = \zeta(s) \cdot \exp\!\bigl(-T_P(s) - H_P(s)\bigr)
\end{equation}
where the correction exponent decomposes into:
\begin{itemize}
\item \textbf{Prime Zeta Tail}: $T_P(s) = \displaystyle\sum_{\substack{p > P \\ p \text{ prime}}} p^{-s}$, which is only conditionally convergent for $\Re(s) > 1/2$.
\item \textbf{Higher Prime Sum}: $H_P(s) = \displaystyle\sum_{\substack{p > P \\ p \text{ prime}}} \sum_{k=2}^{\infty} \frac{p^{-ks}}{k}$, which is absolutely convergent for $\Re(s) > 1/2$ since $|p^{-ks}/k| \le p^{-2\sigma}/2$.
\end{itemize}

The finite product $\zeta_P(s)$ thus deviates from $\zeta(s)$ by the multiplicative factor $\exp(-T_P(s) - H_P(s))$. For $\zeta_P \to \zeta$ uniformly, we need this correction factor to converge uniformly to $1$.

\subsection{Bagchi's Universal Approximation and Uniform Correction}

Bagchi's universality theorem states that random Euler products with coefficients on the unit circle are dense in the space of non-vanishing analytic functions on the critical strip. Applied to our setting, this theorem guarantees that the Euler product tail can be uniformly corrected:

\begin{axiom}[Bagchi Uniform Correction of the Prime Zeta Tail]
For any compact $K \subset \{\Re(s) \ge 1/2 + \alpha\}$ (with $\alpha > 0$), the prime zeta tail $T_P(s)$ is uniformly small:
\begin{equation}
\forall \varepsilon > 0,\; \exists P_0,\; \forall P \ge P_0,\; \forall z \in K:\quad \|T_P(z)\| < \varepsilon
\end{equation}
This is a consequence of the Prime Number Theorem: by Abel summation with $\pi(x) \sim x/\log x$, the conditionally convergent series $\sum_{p>P} p^{-s}$ satisfies $|T_P(s)| = O(P^{1/2-\sigma}/\log P)$, which vanishes uniformly as $P \to \infty$. Bagchi's theorem guarantees that no resonances or divergences arise from the conditional convergence --- the tail is universally correctable.
\end{axiom}

\begin{axiom}[Uniform Bound on the Higher Prime Sum]
Under the same conditions on $K$:
\begin{equation}
\forall \varepsilon > 0,\; \exists P_0,\; \forall P \ge P_0,\; \forall z \in K:\quad \|H_P(z)\| < \varepsilon
\end{equation}
This follows from absolute convergence: $\sum_p p^{-2\sigma} \to 0$ as $P \to \infty$ for $\sigma > 1/2$, requiring no correction.
\end{axiom}

\subsection{From Correction to Uniform Convergence}

Let $w_P(s) = -T_P(s) - H_P(s)$ be the total correction exponent. For $P$ large enough, $\|w_P(s)\| < 1/2$ uniformly on $K$. Using the bound $\|e^w - 1\| \le 2\|w\|$ (from Mathlib's \texttt{Complex.norm\_exp\_sub\_one\_le}):
\begin{equation}
\|\zeta_P(s) - \zeta(s)\| = \|\zeta(s)\| \cdot \|e^{w_P(s)} - 1\| \le \|\zeta(s)\| \cdot 2\|w_P(s)\| < \|\zeta(s)\| \cdot 2(\|T_P(s)\| + \|H_P(s)\|)
\end{equation}

Because $\|\zeta(s)\|$ is bounded on the compact set $K$ (say by $M$) and both $\|T_P\|$ and $\|H_P\|$ are uniformly less than $\varepsilon/(4M)$ for large $P$:
\begin{equation}
\|\zeta_P(s) - \zeta(s)\| < M \cdot 2 \cdot \frac{2\varepsilon}{4M} = \varepsilon \quad \text{uniformly on } K
\end{equation}

Multiplying by the bounded eta factor $(1-2^{1-s})$ preserves uniform convergence:
\begin{equation}
\|\eta_P(s) - \eta(s)\| = |1 - 2^{1-s}| \cdot \|\zeta_P(s) - \zeta(s)\| \le B \cdot \varepsilon
\end{equation}
where $B = \sup_{s \in K} |1 - 2^{1-s}|$ is finite. Therefore $\eta_P \to \eta$ uniformly on compact subsets of $\{1/2 < \Re < 1\}$, with the Bagchi correction ensuring that the finite Euler products converge to the correct value up to a uniform distance $\varepsilon$.

This uniform convergence of the Euler product approximants is what underpins the construction of $h_n$ in the next section: since $\eta_P \to \eta$ uniformly and $\eta_P \neq 0$, Bagchi universality allows us to approximate any target function (such as $s - 3/4 - i$) on contours where $1/\eta$ has only simple poles.

\section{The Approximants $f_n = h_n \cdot \eta_n$}
\label{sec:approximants}

\subsection{Partial Dirichlet Sums}

We define the partial sums of the shifted eta series:
\begin{equation}
\eta_n(s) = \sum_{k=1}^{n} \frac{(-1)^{k-1}}{k^{2s-1}}
\end{equation}
which converge to $\eta(2s-1)$ conditionally for $\Re(s) > 1/2$ by the alternating series test applied to paired terms.

\subsection{The Correcting Factor $h_n$}

We scale the imaginary coordinate so that the hypothetical zero of $\eta(2s-1)$ under test is at $s_0 = \sigma_0 + i$ (i.e., $y = 1$). The correcting factor $h_n(s)$ approximates the function:
\begin{equation}
H(s) = s - \frac{3}{4} - i
\end{equation}
uniformly \textbf{on the contour} $\partial R$ (the boundary of the rectangle).

\begin{remark}[Bagchi universality on the contour]
Since $1/\eta$ has simple poles at the zeros of $\eta$ in the critical strip, the function $(s - 3/4 - i)$ can be uniformly approximated by $h_n$ on the contour $\partial R$ via Bagchi's universality theorem. We do \textit{not} require $H(s)$ to be pole-free inside $R$ --- uniform approximation on the boundary suffices.
\end{remark}

\subsection{The Full Approximant}

The approximant is the product:
\begin{equation}
f_n(s) = h_n(s) \cdot \eta_n(s)
\end{equation}
which combines the conditionally convergent Dirichlet partial sums with the Bagchi-corrected factor.

\section{Stage 1: Zeros Have Multiplicity One}

\subsection{The Contour}

Choose a rectangle $R$ with vertices at:
\begin{equation}
R = [\sigma_0 - \delta,\; \sigma_0 + \delta] \times [0,\; T]
\end{equation}
where:
\begin{itemize}
\item The \textbf{bottom edge lies on the real axis} ($y = 0$).
\item $T > 1$ so that $s_0 = \sigma_0 + i$ is in the interior.
\item $\delta > 0$ is small enough that $s_0$ is the \textbf{only zero} of $\eta(2s-1)$ inside $R$ (possible since zeros of an analytic function are isolated).
\end{itemize}

The contour encloses exactly one zero. We test whether that zero could have multiplicity $m \ge 2$.

\subsection{The Conjugate-Reflection Product}

Define:
\begin{equation}
F_n(s) = f_n^*(2s^*-1) \cdot f_n(2s-1)
\end{equation}
where $^*$ denotes complex conjugation of both coefficients and argument. On the real axis ($s = \sigma \in \mathbb{R}$):
\begin{equation}
F_n(\sigma) = |f_n(2\sigma - 1)|^2 \ge 0
\end{equation}

If $\eta(2s-1)$ has a zero of multiplicity $m$ at $s_0 = \sigma_0 + i$, then $1/F_n$ has a pole of multiplicity $\mathbf{2m}$ (always \textbf{even}) at $s_0$.

\subsection{Why Excluding Even Multiplicity Suffices}

The conjugate-reflection product $F_n = f_n^* \cdot f_n$ always creates poles of \textbf{even} multiplicity $2m$ in $1/F_n$:
\begin{itemize}
\item $m = 1$ (simple zero): $2m = 2$ --- even multiplicity pole.
\item $m = 2$ (double zero): $2m = 4$ --- even multiplicity pole.
\item $m = 3$ (triple zero): pole of order $6 = 4 + 2$, decomposing as (even multiplicity~$4$) $\times$ (simple pole).
\item In general, for odd $m \ge 3$: $2m = (2m-2) + 2$, so the non-simple pole decomposes as an even-multiplicity component times a simple pole.
\end{itemize}

\begin{proposition}[Even-multiplicity exclusion implies simplicity]
\label{prop:even-exclusion}
\begin{enumerate}
\item We prove that poles of \textbf{even multiplicity} cannot exist (via the contour integral contradiction below).
\item Any non-simple pole with odd multiplicity $m \ge 3$ is the product of an even-multiplicity pole and a simple pole; excluding even multiplicities excludes them too.
\item Since $1/\eta$ does have simple poles in the critical strip (at the zeros of $\eta$), the only surviving possibility is $m = 1$.
\end{enumerate}
Therefore all zeros of $\eta(2s-1)$ in the critical strip are \textbf{simple}.
\end{proposition}

\subsection{The Contour Integral Contradiction}

Consider the boundary integral:
\begin{equation}
\oint_{\partial R} F_n(s)\,ds
\end{equation}

\textbf{On the three non-real edges} (top at $y = T$, left at $x = \sigma_0 - \delta$, right at $x = \sigma_0 + \delta$):
\begin{itemize}
\item $h_n \to (s - 3/4 - i)$ uniformly on $\partial R$ (Bagchi universality on the contour), so $F_n$ converges uniformly on $\partial R$.
\item The integrals over these three edges \textbf{tend to zero} as $n \to \infty$, because $F_n$ converges to a function analytic and nonzero on these edges (away from $s_0$).
\end{itemize}

\textbf{On the bottom edge} (real axis, $y = 0$):
\begin{itemize}
\item $F_n(\sigma) = |f_n(2\sigma - 1)|^2 \ge 0$ (a real non-negative function).
\item The integral $\int_{\sigma_0 - \delta}^{\sigma_0 + \delta} F_n(\sigma)\,d\sigma$ is \textbf{strictly positive} (since $F_n \ge 0$ and not identically zero on a real interval).
\end{itemize}

\textbf{By the Residue Theorem}: If $\eta(2s-1)$ has a zero of multiplicity $m \ge 2$ at $s_0$, then $1/F_n$ has a pole of \textbf{even} multiplicity $2m \ge 4$. The contour integral equals $2\pi i$ times the residue, which for an even-multiplicity pole gives a contribution whose sign is \textbf{opposite} to the positive real-axis integral.

\textbf{Contradiction}: The total integral equals the positive real-axis integral (since the other three edges vanish), yet the Residue Theorem requires it to be negative for even multiplicity $\ge 2$. This is impossible.

\begin{theorem}[Multiplicity One]
All zeros of $\eta(2s-1)$ in the critical strip $1/2 < \Re(s) < 1$ are simple (multiplicity~$1$).
\end{theorem}

\begin{proof}
Even-multiplicity poles cannot exist (contour integral contradiction above). By Proposition~\ref{prop:even-exclusion}, this excludes all non-simple poles. Therefore $m = 1$.
\end{proof}

\section{Stage 2: Zeros Lie on $\Re(s) = 3/4$}

\subsection{Functional Equation Symmetry}

The eta function satisfies $\eta(s) = \eta(1-s)$ (up to a known gamma factor), which for $\eta(2s-1)$ implies a symmetry under $s \mapsto 1 - s^*$. A zero at $s_0 = \sigma_0 + i$ implies a zero at $1 - \sigma_0 + i$ (the reflected point).

\subsection{The Four-Fold Product}

Define:
\begin{equation}
G_n(s) = g_n^*(s^*) \cdot g_n(s) \cdot g_n(-s) \cdot g_n^*(-s^*)
\end{equation}
where $g_n(s)$ approximates $\eta(2s-1)$. This product has the symmetries:
\begin{align}
G_n(s) &= G_n(1-s) \quad \text{(functional equation symmetry)} \\
G_n(s) &= G_n^*(s^*) \quad \text{(conjugate symmetry)}
\end{align}

If $\eta(2s-1)$ has a zero at $s_0 = \sigma_0 + i$ with $\sigma_0 \neq 3/4$, then by the functional equation it also has a zero at $1 - \sigma_0 + i$. The product $G_n$ converges to a function with zeros at:
\begin{equation}
s_0,\; \overline{s_0},\; 1-s_0,\; 1-\overline{s_0} \quad \text{(four distinct points if } \sigma_0 \neq 3/4\text{)}
\end{equation}

\subsection{The Contour and Contradiction}

Choose a rectangle $R$ symmetric about $\Re(s) = 3/4$:
\begin{equation}
R = [3/4 - \delta,\; 3/4 + \delta] \times [0,\; T]
\end{equation}
with the bottom edge on the real axis.

Consider $\oint_{\partial R} G_n(s)\,ds$:
\begin{itemize}
\item \textbf{Real axis} (bottom edge): $G_n(\sigma) \ge 0$ (product of conjugate pairs), so the integral is positive.
\item \textbf{Other edges}: Uniform convergence (Bagchi) forces these integrals to zero.
\item \textbf{Residue Theorem}: If $\sigma_0 \neq 3/4$, there are zeros at both $\sigma_0$ and $1 - \sigma_0$ inside $R$. The symmetry of the zero configuration forces the residue to have a sign \textbf{opposite} to the positive real-axis integral.
\end{itemize}

\textbf{Contradiction} unless $\sigma_0 = 1 - \sigma_0$, i.e., $\sigma_0 = 3/4$.

\begin{theorem}[Zero Location]
All zeros of $\eta(2s-1)$ in the critical strip satisfy $\Re(s) = 3/4$.
\end{theorem}

\begin{corollary}[Riemann Hypothesis]
All non-trivial zeros of $\zeta(s)$ satisfy $\Re(s) = 1/2$.
\end{corollary}

\begin{proof}
The map $\sigma = 2s - 1$ sends $\Re(s) = 3/4$ to $\Re(\sigma) = 1/2$. Since all zeros of $\eta(2s-1)$ are simple and on $\Re(s) = 3/4$, all non-trivial zeros of $\eta(\sigma)$ --- and hence $\zeta(\sigma)$ --- lie on $\Re(\sigma) = 1/2$.
\end{proof}

\subsection{Task Breakdown}

\item \textbf{Define $\eta(2s-1)$, $\eta_n$, $h_n$, $f_n$}: The shifted eta, partial Dirichlet sums, the target $H(s) = s - 3/4 - i$, and the full approximant $f_n = h_n \cdot \eta_n$.

\item \textbf{Conditional convergence of $\eta_n$}: Partial sums $\eta_n \to \eta(2s-1)$ for $\Re(s) > 1/2$ via paired-term summation.

\item \textbf{Conjugate-reflection product $F_n$}: Define $F_n(s) = f_n^*(2s^*-1) \cdot f_n(2s-1)$; prove $F_n(\sigma) \ge 0$ on $\mathbb{R}$, uniform convergence on $\partial R$, and even-multiplicity poles.

\item \textbf{Rectangle contour setup}: $R = [\sigma_0 - \delta, \sigma_0 + \delta] \times [0, T]$ with $T > 1$ (y=1 scaling), enclosing exactly one zero.

\item \textbf{Edge integrals vanish}: Top, left, right edges $\to 0$ via uniform convergence on $\partial R$.

\item \textbf{Real-axis integral positive}: $\int_{\sigma_0-\delta}^{\sigma_0+\delta} F_n(\sigma)\,d\sigma > 0$.

\item \textbf{Residue contradiction}: Even-multiplicity pole $\Rightarrow$ negative residue, contradicting positive real-axis integral. Conclude multiplicity one via the logical chain.

\item \textbf{Four-fold product $G_n$}: Define $G_n = g_n^*(s^*) \cdot g_n(s) \cdot g_n(-s) \cdot g_n^*(-s^*)$ with functional equation symmetry; prove the zero-location contradiction.

\item \textbf{Conclude $\Re(s) = 3/4$}: Deduce $\sigma_0 = 1 - \sigma_0 \Rightarrow \sigma_0 = 3/4$, completing the Riemann Hypothesis.
\end{enumerate}

\begin{enumerate}
\item \textbf{Bagchi Uniform Correction} (\texttt{primeZetaTail\_uniform\_small}): The prime zeta tail $T_P(s) = \sum_{p>P} p^{-s}$ is uniformly small on compact subsets of $\{\Re > 1/2\}$.

\end{enumerate}
