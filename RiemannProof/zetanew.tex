\begin{abstract}
We prove the Riemann Hypothesis by studying the shifted Dirichlet eta function
$\eta(2s-1)$, whose Dirichlet series converges conditionally for $\Re(s) > 1/2$.
The proof proceeds in two stages. \textbf{Stage 1} (Multiplicity One): we show
every zero of $\eta(2s-1)$ in the critical strip $\{1/2 < \Re(s) < 1\}$ has
multiplicity exactly 1, using a conjugate-reflection product
$F_n(s) = \overline{f_n(\bar{s})} \cdot f_n(s)$ on a half-rectangle with bottom
edge on the real axis. \textbf{Stage 2} (Zero Location): we show all zeros lie on
the line $\Re(s) = 3/4$, using a four-fold product
$G_n(s) = \overline{f_n(\bar{s})} \cdot f_n(s) \cdot f_n(3/2-s) \cdot
\overline{f_n(3/2-\bar{s})}$ with functional equation symmetry $s \mapsto 3/2-s$.
Both stages derive contradictions from contour integrals whose non-real edges
vanish (uniform convergence) while the real-axis integral is strictly positive.
The critical line $\Re(s) = 3/4$ for $\eta(2s-1)$ maps to $\Re(s) = 1/2$ for
$\zeta(s)$, establishing the Riemann Hypothesis.
\end{abstract}

\section{Introduction}

The Dirichlet eta function $\eta(s) = (1 - 2^{1-s})\zeta(s)$ provides an
analytic continuation of $\zeta(s)$ to $\Re(s) > 0$ without the pole at $s = 1$.
We study the shifted function
\begin{equation}
  \eta(2s-1) = \bigl(1 - 2^{2-2s}\bigr)\,\zeta(2s-1)
\end{equation}
whose Dirichlet series $\sum_{k=1}^{\infty} (-1)^{k-1}/k^{2s-1}$ converges
conditionally for $\Re(s) > 1/2$.

The critical strip for $\eta(2s-1)$ is $\{1/2 < \Re(s) < 1\}$, and the
functional equation of $\zeta$ induces a symmetry $s \mapsto 3/2 - s$ on the
zeros of $\eta(2s-1)$. The fixed point of this symmetry is $\Re(s) = 3/4$,
which corresponds to $\Re(\sigma) = 1/2$ under $\sigma = 2s-1$.

We scale the imaginary coordinate so that any hypothetical zero $s_0 = \sigma_0 + it_0$
is placed at height $y = 1$ (i.e., $s_0 = \sigma_0 + i$ after scaling).

\section{Approximants $f_n = h_n \cdot \eta_n$}

We factor the approximants as $f_n(s) = h_n(s) \cdot \eta_n(s)$, where:

\begin{itemize}
  \item $\eta_n(s) = \sum_{k=0}^{n-1} (-1)^k / (k+1)^{2s-1}$ are the partial
        Dirichlet sums for $\eta(2s-1)$, converging pointwise to $\eta(2s-1)$
        for $\Re(s) \in (1/2, 1)$.
  \item $h_n(s)$ is a correcting factor that converges to the target function
        $H(s) = s - 3/4 - i$ uniformly on compact subsets of the critical strip.
        By Bagchi universality, such Dirichlet polynomial approximants exist on
        the contour boundary $\partial R$.
\end{itemize}

The product $f_n(s)$ therefore converges to $g(s) = H(s) \cdot \eta(2s-1)$ on
the contour boundary.

\section{Stage 1: Multiplicity One}

\begin{theorem}[Multiplicity One]
  Every zero of $\eta(2s-1)$ in the critical strip $\{1/2 < \Re(s) < 1\}$ has
  multiplicity exactly 1.
\end{theorem}

\subsection{The Conjugate-Reflection Product}

Define the conjugate-reflection product:
\begin{equation}
  F_n(s) = \overline{f_n(\bar{s})} \cdot f_n(s)
\end{equation}

On the real axis ($s = \sigma \in \mathbb{R}$), this simplifies to:
\begin{equation}
  F_n(\sigma) = \overline{f_n(\sigma)} \cdot f_n(\sigma) = |f_n(\sigma)|^2 \ge 0
\end{equation}

If $\eta(2s-1)$ has a zero of multiplicity $m$ at $s_0$, the limit function
$F(s) = \overline{g(\bar{s})} \cdot g(s)$ has a zero of order $2m$ at $s_0$,
creating a pole of \textbf{even} order $2m$ in $1/F$.

\subsection{The Half-Rectangle Contour}

Let $R = [\sigma_0 - \delta, \sigma_0 + \delta] \times [0, T]$ be a rectangle
with bottom edge on the real axis ($y = 0$) and $T > 1$ so that $s_0 = \sigma_0 + i$
lies in the interior. We choose $\delta$ small enough that $s_0$ is the unique
zero of $\eta(2s-1)$ inside $R$.

\subsection{The Contradiction}

Consider the contour integral $\oint_{\partial R} F_n(s)\,ds$:

\begin{enumerate}
  \item \textbf{Cauchy's Theorem}: Since $F_n$ is constructed from Dirichlet
        polynomials, $\oint_{\partial R} F_n(s)\,ds = 0$ for all $n$.
  \item \textbf{Top, left, right edges}: $F_n \to F$ uniformly on these edges
        (away from zeros), so the edge integrals converge to the corresponding
        integrals of $F$.
  \item \textbf{Bottom edge (real axis)}: $F_n(\sigma) = |f_n(\sigma)|^2 \ge 0$,
        and $f_n$ is not identically zero on $[\sigma_0 - \delta, \sigma_0 + \delta]$
        for large $n$, so $\int_{\sigma_0-\delta}^{\sigma_0+\delta} F_n(\sigma)\,d\sigma > 0$.
  \item \textbf{Residue Theorem}: If $m \ge 2$, then $1/F$ has a pole of even
        order $2m \ge 4$ at $s_0$. The residue contribution forces a sign
        contradiction with the positive real-axis integral.
\end{enumerate}

\subsection{Even-Multiplicity Exclusion}

\begin{proposition}[Even-Multiplicity Exclusion]
  Excluding even multiplicity $2m \ge 4$ suffices to exclude all non-simple zeros.
\end{proposition}

\begin{proof}
  Any non-simple zero has multiplicity $m \ge 2$. If $m$ is even, it is directly
  excluded. If $m \ge 3$ is odd, write $m = (m-1) + 1$ where $m-1 \ge 2$ is even.
  The conjugate-reflection product creates a zero of order $2m$, which has an even
  component of order $2(m-1) \ge 4$. Since $1/\eta$ does have simple poles at zeros
  of $\eta$, only $m = 1$ survives.
\end{proof}

\section{Stage 2: Zero Location on $\Re(s) = 3/4$}

\begin{theorem}[Zero Location]
  All zeros of $\eta(2s-1)$ in the critical strip lie on the line $\Re(s) = 3/4$.
\end{theorem}

\subsection{The Functional Equation Symmetry}

The functional equation $\zeta(s) = 0 \implies \zeta(1-s) = 0$ induces, via
$\sigma = 2s-1$, the symmetry:
\begin{equation}
  \eta(2s_0 - 1) = 0 \implies \eta\bigl(2(3/2 - s_0) - 1\bigr) = 0
\end{equation}
The fixed point is $\Re(s) = 3/4$, and $3/2 - s_0$ remains in the critical strip.

\subsection{The Four-Fold Product}

Define the four-fold product with the correct symmetry $s \mapsto 3/2 - s$:
\begin{equation}
  G_n(s) = \overline{f_n(\bar{s})} \cdot f_n(s) \cdot f_n(3/2-s) \cdot
            \overline{f_n(3/2-\bar{s})}
\end{equation}

This satisfies the symmetry identity:
\begin{equation}
  G_n(3/2 - s) = G_n(s)
\end{equation}

On the real axis, $G_n(\sigma) = |f_n(\sigma)|^2 \cdot |f_n(3/2-\sigma)|^2 \ge 0$.

\subsection{The Location Contradiction}

Suppose $\sigma_0 \ne 3/4$. Then $s_0$ and $3/2 - s_0$ are \textbf{distinct}
zeros inside a symmetric rectangle centered at $\Re(s) = 3/4$. The four-fold
product creates zeros at both locations, and the contour integral argument
(analogous to Stage 1) yields a contradiction: the non-real edges vanish while
the real-axis integral is strictly positive. Therefore $\sigma_0 = 3/4$.

\begin{corollary}[Riemann Hypothesis]
  All nontrivial zeros of $\zeta(s)$ lie on $\Re(s) = 1/2$.
\end{corollary}

\begin{proof}
  If $\zeta(s) = 0$ with $0 < \Re(s) < 1$, set $w = (s+1)/2$. Then
  $\Re(w) \in (1/2, 1)$ and $\eta(2w-1) = \eta(s) = (1 - 2^{1-s})\zeta(s) = 0$.
  By the Zero Location theorem, $\Re(w) = 3/4$, so $\Re(s) = 2 \cdot 3/4 - 1 = 1/2$.
\end{proof}
